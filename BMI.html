def calculate_body_metrics(age=42, gender='male', weight_kg=80.0, height_cm=165.0, activity_level='light', goal='lose_weight'): """ Hàm tính toán BMI, BMR, TDEE và tỷ lệ thực đơn. activity_level: 'sedentary' (ít vận động), 'light' (vận động nhẹ), 'moderate' (vừa), 'active' (nhiều) goal: 'maintain' (giữ cân), 'lose_weight' (giảm cân), 'gain_weight' (tăng cân) """ # 1. Tính Chỉ số khối cơ thể (BMI) height_m = height_cm / 100 bmi = weight_kg / (height_m ** 2) # 2. Tính Tỷ lệ trao đổi chất cơ bản (BMR) - Công thức Mifflin-St Jeor if gender.lower() == 'male': bmr = (10 * weight_kg) + (6.25 * height_cm) - (5 * age) + 5 else: bmr = (10 * weight_kg) + (6.25 * height_cm) - (5 * age) - 161 # 3. Tính Tổng năng lượng tiêu hao (TDEE) dựa trên mức độ vận động activity_multipliers = { 'sedentary': 1.2, # Ít vận động (làm việc văn phòng) 'light': 1.375, # Vận động nhẹ (đi lại nhiều, tập thể dục nhẹ 1-3 ngày/tuần) 'moderate': 1.55, # Vận động vừa (tập luyện 3-5 ngày/tuần) 'active': 1.725, # Vận động cường độ cao (6-7 ngày/tuần) } tdee = bmr * activity_multipliers.get(activity_level, 1.2) # 4. Xác định Mục tiêu Calo target_calories = tdee if goal == 'lose_weight': target_calories -= 500 # Cắt giảm 500 calo/ngày để giảm mỡ an toàn elif goal == 'gain_weight': target_calories += 500 # Thêm 500 calo/ngày để tăng cân # 5. Phân bổ Tỷ lệ Dinh dưỡng (Macro) # Áp dụng tỷ lệ chuẩn để cân bằng vóc dáng: 40% Tinh bột, 30% Đạm, 30% Chất béo # Hệ số quy đổi: 1g Carb = 4 calo | 1g Protein = 4 calo | 1g Fat = 9 calo carb_cals = target_calories * 0.40 protein_cals = target_calories * 0.30 fat_cals = target_calories * 0.30 carbs_g = carb_cals / 4 protein_g = protein_cals / 4 fat_g = fat_cals / 9 # === IN KẾT QUẢ === print("="*50) print(" BÁO CÁO CHỈ SỐ CƠ THỂ VÀ THỰC ĐƠN DINH DƯỠNG ") print("="*50) print(f"\n[1] CHỈ SỐ CƠ THỂ (Cân nặng: {weight_kg}kg, Chiều cao: {height_cm}cm, Tuổi: {age})") print(f"- Chỉ số BMI: {bmi:.2f}") if bmi >= 25: print(" -> Đánh giá: Thừa cân (Nên ưu tiên mục tiêu giảm mỡ)") elif bmi >= 18.5: print(" -> Đánh giá: Cân nặng bình thường") else: print(" -> Đánh giá: Thiếu cân") print(f"\n[2] NĂNG LƯỢNG TIÊU HAO") print(f"- Tỷ lệ trao đổi chất cơ bản (BMR): {bmr:.0f} calo/ngày (Năng lượng tối thiểu để duy trì sự sống)") print(f"- Tổng năng lượng tiêu hao (TDEE): {tdee:.0f} calo/ngày") print(f" -> Mức calo cần nạp mỗi ngày ({goal}): {target_calories:.0f} calo") print(f"\n[3] TỶ LỆ THỰC ĐƠN VÀ THỰC PHẨM KHUYẾN NGHỊ (40% - 30% - 30%)") print(f"- Tinh bột (Carbohydrate): {carbs_g:.0f}g/ngày ({carb_cals:.0f} calo)") print(f" -> Nguồn ưu tiên: Gạo lứt, khoai lang, yến mạch, bánh mì nguyên cám.") print(f"- Đạm (Protein): {protein_g:.0f}g/ngày ({protein_cals:.0f} calo)") print(f" -> Nguồn ưu tiên: Ức gà, cá biển, thịt nạc, trứng, sữa non bổ sung kháng thể, các loại đậu.") print(f"- Chất béo (Fat): {fat_g:.0f}g/ngày ({fat_cals:.0f} calo)") print(f" -> Nguồn ưu tiên: Dầu olive, quả bơ, cá hồi, các loại hạt (hạnh nhân, óc chó).") print("\n*Lưu ý: Đừng quên bổ sung nhiều rau xanh (chất xơ) và uống đủ 2-2.5 lít nước mỗi ngày.") # Thực thi hàm calculate_body_metrics()
